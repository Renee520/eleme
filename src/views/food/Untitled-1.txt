class Person {
  constructor(firstName, lastName) {
    this.firstName = firstName;
    this.lastName = lastName;
  }
}

let p = new Person('111', '222');w
Person.prototype.getFullName = () => this.firstName + this.lastName; // 调用p.getFullName()，这里this指window是吧嗯，也就是undefined + undefined
console.log(p.getFullName())

// 输出啊啥undefined
其实是typeError，p是Person实例，实例调用的方法其实是Person原型上的方法，但是class Person里并没有这个方法，除非Person.prototype.getFullName定义了，否则
实例中无法调用此方法

var obj = {

    console.log(this.a);
  },

    console.log(this.a); //知道这里为嘛是undefined吗箭头函数的this关联父级作用域的this，即obj的this，即window
  }
}
